private void jSpinner1StateChanged(javax.swing.event.ChangeEvent evt) {                                       
        // TODO add your handling code here:
        SpinnerModel sm = jSpinner1.getModel();
        int gewuenschteAnzahl = (int) sm.getValue();
        int aktuelleAnzahl = aktuellerJahrgang.getKlassenanzahl();
        while( gewuenschteAnzahl> aktuelleAnzahl){
            aktuellerJahrgang.setKlassenanzahl(aktuelleAnzahl++);
            char buchstabe = (char) (aktuelleAnzahl+96);
            Klasse k = new Klasse(""+buchstabe,aktuellerJahrgang.getJahrgang());
            jPanelKlassen.add(new GruppenPanel(k.getSchueler(), aktuellerJahrgang.getJahrgang() + k.getBuchstabe()));
            jPanelKlassen.revalidate();
            jPanelKlassen.repaint();
        }
    }
